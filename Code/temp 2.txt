public /*abstract*/ void add(T data) 
    {
        Node<T> newNode = new Node<>();
        newNode.data = data;
        newNode.next = null;
        
        if (head == null) 
        {
            head = newNode;
        } 
        else 
        {
            Node<T> current = head;
            while (current.next != null) 
            {
                current = current.next;
            }
            current.next = newNode;
        }
        
        length++;
    }
    
    public /*abstract*/ void delete()
    {
        if (head == null) 
        {
            return;
        }
        
        if (head.next == null) 
        {
            head = null;
        } 
        else 
        {
            Node<T> current = head;
            while (current.next.next != null) 
            {
                current = current.next;
            }
            current.next = null;
        }
        
        length--;
    }
